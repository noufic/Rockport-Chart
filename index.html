<head>

  <meta charset="UTF-8" />

  <title>Animated Bar Chart Example</title>

  <!-- Include Chart.js from a CDN -->

  <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

  <style>

    /* Optional styling */

    body {

      font-family: Arial, sans-serif;

      margin: 20px;

      background: #f9f9f9;

    }

    .chart-container {

      width: 600px; /* Adjust as needed */

      margin: 0 auto;

    }

  </style>

</head>


<body>

  <!-- Section where you can place your descriptive text on the left in your layout -->


  <div class="chart-container">

    <canvas id="myChart" width="600" height="400"></canvas>

  </div>


  <script>

    // Your data for Rockport vs. State

    // Example categories: 

    // 1) HH Income <$50,000, 

    // 2) Income <200% of Fed. Poverty Level, 

    // 3) Households with Public Assistance Income

    const chartLabels = [

      "HH Income <$50,000",

      "Income <200% of Fed. Poverty Level",

      "Households w/ Public Assistance",

    ];


    // Rockport percentages

    const rockportData = [25, 17, 12];


    // Statewide percentages

    const stateData = [37, 27, 21];


    // Get reference to canvas

    const ctx = document.getElementById("myChart").getContext("2d");


    // Create the bar chart

    new Chart(ctx, {

      type: "bar",

      data: {

        labels: chartLabels,

        datasets: [

          {

            label: "Rockport",

            data: rockportData,

            backgroundColor: "rgba(115, 140, 93, 0.7)", // Adjust color as needed

          },

          {

            label: "State",

            data: stateData,

            backgroundColor: "rgba(184, 162, 120, 0.7)", // Adjust color as needed

          },

        ],

      },

      options: {

        responsive: true,

        animation: {

          // Animate on load

          duration: 2000,

          easing: "easeOutBounce",

        },

        scales: {

          y: {

            beginAtZero: true,

            ticks: {

              // Append '%' to each tick

              callback: function (value) {

                return value + "%";

              },

            },

            title: {

              display: true,

              text: "Percentage of Households",

            },

          },

          x: {

            title: {

              display: true,

              text: "Income & Employment Indicators",

            },

          },

        },

        plugins: {

          legend: {

            display: true,

          },

          tooltip: {

            callbacks: {

              label: function (context) {

                // Show the label and percentage with a '%' symbol

                let label = context.dataset.label || "";

                let value = context.parsed.y || 0;

                return label + ": " + value + "%";

              },

            },

          },

        },

      },

    });

  </script>

</body>
